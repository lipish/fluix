Installation

Add Fluix to your Cargo.toml:

[dependencies]
fluix = "0.1.21"
gpui = "0.2"

Quick Start

use gpui::*;
use fluix::*;

fn main() {
    let app = Application::new()
        .with_assets(fluix::Assets);  // Important! Load SVG icons

    app.run(move |cx| {
        cx.open_window(WindowOptions::default(), |window, cx| {
            cx.new(|cx| MyView::new(window, cx))
        }).unwrap();
    });
}

struct MyView {
    button: Entity<Button>,
}

impl MyView {
    fn new(window: &mut Window, cx: &mut Context<Self>) -> Self {
        let button = cx.new(|_cx| {
            Button::new("Click Me")
                .variant(ButtonVariant::Primary)
                .size(ComponentSize::Medium)
        });
        
        cx.subscribe_in(&button, window, Self::on_click).detach();
        
        Self { button }
    }
    
    fn on_click(&mut self, _: &Entity<Button>, _: &ButtonEvent, 
                _: &mut Window, _: &mut Context<Self>) {
        println!("Button clicked!");
    }
}

impl Render for MyView {
    fn render(&mut self, _: &mut Window, _: &mut Context<Self>) -> impl IntoElement {
        div()
            .p_4()
            .child(self.button.clone())
    }
}

Button

Interactive button component with multiple variants and sizes.

Variants

Button::new("Primary")
    .variant(ButtonVariant::Primary)

Button::new("Secondary")
    .variant(ButtonVariant::Secondary)

Button::new("Outline")
    .variant(ButtonVariant::Outline)

Button::new("Text")
    .variant(ButtonVariant::Text)

Button::new("Danger")
    .variant(ButtonVariant::Danger)

Sizes

Button::new("Small")
    .size(ComponentSize::Small)

Button::new("Medium")
    .size(ComponentSize::Medium)  // Default

Button::new("Large")
    .size(ComponentSize::Large)

Properties

Property Type Default Description
variant ButtonVariant Primary Visual style variant
size ComponentSize Medium Button size
disabled bool false Disable button interaction
loading bool false Show loading state
full_width bool false Make button full width

Events

pub enum ButtonEvent {
    Click,  // Button was clicked
}

TextInput

Single-line text input component with IME support and validation.

Usage

let input = cx.new(|cx| {
    TextInput::new(cx)
        .placeholder("Enter your name...")
        .max_length(50)
});

cx.subscribe_in(&input, window, |_, _, event: &TextInputEvent, _, _| {
    match event {
        TextInputEvent::Change(value) => println!("Value: {}", value),
        TextInputEvent::Submit(value) => println!("Submitted: {}", value),
        _ => {}
    }
}).detach();

Properties

Property Type Default Description
placeholder String "" Placeholder text
value String "" Initial value
disabled bool false Disable input
is_password bool false Password mode
max_length Option<usize> None Maximum length

TextArea

Multi-line text editor component with auto-height and scroll support.

Usage

let textarea = cx.new(|cx| {
    TextArea::new(cx)
        .placeholder("Type your message...")
        .min_height(80.0)
        .max_height(200.0)
});

Select

Dropdown selection component supporting single and multiple selection.

Usage

let select = cx.new(|cx| {
    Select::new(cx)
        .placeholder("Choose an option...")
        .options(vec![
            SelectOption::new("react", "React"),
            SelectOption::new("vue", "Vue"),
            SelectOption::new("angular", "Angular"),
        ])
        .multiple(false)
});
